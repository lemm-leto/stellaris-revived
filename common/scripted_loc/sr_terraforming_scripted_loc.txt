defined_text = {
	name = GetRadiation
	text = {
		trigger = {
			check_variable = { which = radiation value <= 5.0 }
    }
		localization_key = "sr_terraforming.window.planet.radiation.low"
	}
	text = {
		trigger = {
			AND = {
				check_variable = { which = radiation value > 5.0 }
				check_variable = { which = radiation value <= 25.0 }
			}
    }
		localization_key = "sr_terraforming.window.planet.radiation.medium"
	}
	text = {
		trigger = {
			check_variable = { which = radiation value > 25.0 }
    }
		localization_key = "sr_terraforming.window.planet.radiation.high"
	}
}
